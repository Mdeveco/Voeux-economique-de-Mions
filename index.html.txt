<script>
const SHEET_URL = "COLLE_ICI_L_URL_CSV";

let displayed = new Set();

async function fetchData() {
    const response = await fetch(SHEET_URL + "&cache=" + Date.now());
    const text = await response.text();
    const rows = text.split("\n").slice(1);

    rows.forEach(row => {
        const cols = row.split(",");
        if (cols.length < 4) return;

        const id = row;
        if (displayed.has(id)) return;

        displayed.add(id);

        const card = document.createElement("div");
        card.className = "card";

        card.innerHTML = `
            <h2>${cols[1]}</h2>
            <h3>${cols[2]}</h3>
            <div class="badge">${cols[3]}</div>
        `;

        document.getElementById("wall").prepend(card);

        if (document.getElementById("wall").children.length > 18) {
            document.getElementById("wall").lastChild.remove();
        }
    });
}

setInterval(fetchData, 3000);
</script>
